            const tooltipTitleCallback = (tooltipItems) => {

                if (Array.isArray(label)) { return label.join(' '); }

                return label;

            };

                responsive: true,

                maintainAspectRatio: false,

                plugins: {

                        labels: { color: '#8892b0' }

                    },

                    tooltip: {

                        callbacks: { title: tooltipTitleCallback }

                    }

                },

            };


            const createEfficiencyChart = () => {

                const ctx = document.getElementById('efficiencyChart')?.getContext('2d');

                    type: 'doughnut',

                    data: {

                    },

                    options: { ...chartOptions, cutout: '70%' }

                });

            };


            const createChurnChart = () => {

                const ctx = document.getElementById('churnChart')?.getContext('2d');

            const observerCallback = (entries, observer) => {

                entries.forEach(entry => {

                    if (entry.isIntersecting) {

                        entry.target.classList.add('is-visible');

                    }

                });

            };

            

            const observer = new IntersectionObserver(observerCallback, { rootMargin: '0px', threshold: 0.1 });

            document.querySelectorAll('.fade-in-section').forEach(el => observer.observe(el));

            

            const chartObserver = new IntersectionObserver((entries, chartObs) => {

                entries.forEach(entry => {

                        chartObs.unobserve(entry.target);

                });

            }, {threshold: 0.5});


            const efficiencyCanvas = document.getElementById('efficiencyChart');

            const churnCanvas = document.getElementById('churnChart');
